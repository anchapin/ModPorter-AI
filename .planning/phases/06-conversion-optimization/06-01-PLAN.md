def extract_classes(tree) -> List[Dict]:
    """Extract all class declarations from AST."""
    classes = []
    
    def traverse(node):
        if node.type == "class_declaration":
            class_info = {
                "name": get_class_name(node),
                "modifiers": get_modifiers(node),
                "superclass": get_superclass(node),
                "interfaces": get_interfaces(node),
            }
            classes.append(class_info)
        
        for child in node.children:
            traverse(child)
    
    traverse(tree.root_node)
    return classes
```

class JavaAnalyzerAgent:
    def __init__(self):
        # New: Tree-sitter parser
        self.parser = TreeSitterJavaParser()
        # Keep javalang as fallback
        self.fallback_parser = JavalangParser()
    
    def analyze_mod_file(self, mod_path: str) -> str:
        try:
            # Try tree-sitter first
            result = self.parser.analyze(mod_path)
            return result
        except TreeSitterError as e:
            # Fallback to javalang
            logger.warning(f"Tree-sitter failed, using fallback: {e}")
            return self.fallback_parser.analyze(mod_path)
```

    """Parse code with error recovery."""
    tree = parser.parse(code)
    
    # Check for error nodes
    error_nodes = find_error_nodes(tree)
    
    if error_nodes:
        # Try to recover
        recovered = attempt_recovery(tree, error_nodes)
        return ParseResult(
            tree=recovered,
            errors=[str(e) for e in error_nodes],
            partial=True,
        )
    
    return ParseResult(tree=tree, errors=[], partial=False)
