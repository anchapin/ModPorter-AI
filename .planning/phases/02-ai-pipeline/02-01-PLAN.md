**Implementation Notes**:

```python
# backend/src/services/java_parser.py
import tree_sitter_java as ts_java
from tree_sitter import Language, Parser

# Build language parser
JAVA_LANGUAGE = Language(ts_java.language())
parser = Parser()
parser.set_language(JAVA_LANGUAGE)

def parse_java_file(source_code: str) -> tree_sitter.Tree:
    """Parse Java source code and return AST"""
    return parser.parse(bytes(source_code, "utf8"))
```

```python
def extract_classes(tree) -> List[dict]:
    """Extract all class declarations from AST"""
    classes = []
    
    def traverse(node):
        if node.type == "class_declaration":
            class_info = {
                "name": get_class_name(node),
                "modifiers": get_modifiers(node),
                "superclass": get_superclass(node),
                "interfaces": get_interfaces(node),
                "methods": [],
                "fields": [],
            }
            classes.append(class_info)
        
        for child in node.children:
            traverse(child)
    
    traverse(tree.root_node)
    return classes
```

```python
class DataFlowGraph:
    def __init__(self):
        self.nodes = {}  # node_id -> node_info
        self.edges = []  # (source_id, target_id, edge_type)
    
    def add_variable(self, name: str, type: str, scope: str):
        """Add variable definition node"""
        node_id = f"var:{name}:{scope}"
        self.nodes[node_id] = {
            "type": "variable",
            "name": name,
            "data_type": type,
            "scope": scope,
        }
    
    def add_usage(self, var_name: str, usage_location: str):
        """Add variable usage edge"""
        var_id = f"var:{var_name}"
        usage_id = f"use:{usage_location}"
        self.edges.append((var_id, usage_id, "USED_AT"))
```

**Detection Logic**:
```python
def detect_mod_loader(jar_contents: dict) -> str:
    """Detect mod loader from JAR contents"""
    
    # Check metadata files
    if "META-INF/mods.toml" in jar_contents:
        return "forge"
    if "fabric.mod.json" in jar_contents:
        return "fabric"
    if "neoforge.mod.json" in jar_contents:
        return "neoforge"
    if "quilt.mod.json" in jar_contents:
        return "quilt"
    
    # Check imports in Java files
    for source in jar_contents.get("sources", []):
        if "net.minecraftforge" in source:
            return "forge"
        if "net.fabricmc" in source:
            return "fabric"
        if "net.neoforged" in source:
            return "neoforge"
    
    return "unknown"
```
