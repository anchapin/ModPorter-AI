```python
# modal_deployment.py
import modal
from transformers import AutoModelForSeq2SeqLM, AutoTokenizer

stub = modal.Stub("codet5-plus")

@stub.cls(
    gpu="A10G",
    timeout=600,
    memory=16384,
)
class CodeT5Plus:
    def __enter__(self):
        self.tokenizer = AutoTokenizer.from_pretrained("Salesforce/codet5p-16b")
        self.model = AutoModelForSeq2SeqLM.from_pretrained("Salesforce/codet5p-16b")
    
    @modal.method()
    def translate(self, java_code: str) -> str:
        inputs = self.tokenizer(java_code, return_tensors="pt", max_length=512, truncation=True)
        outputs = self.model.generate(**inputs, max_length=512, num_beams=10)
        return self.tokenizer.decode(outputs[0], skip_special_tokens=True)
```

```python
# ai_engine/services/deepseek_client.py
import openai  # DeepSeek uses OpenAI-compatible API

class DeepSeekClient:
    def __init__(self, api_key: str):
        self.client = openai.OpenAI(
            api_key=api_key,
            base_url="https://api.deepseek.com/v1"
        )
        self.model = "deepseek-coder"
    
    def translate(self, java_code: str, context: list = None) -> str:
        messages = [
            {"role": "system", "content": "You are a Java to JavaScript translator."},
            {"role": "user", "content": f"Translate this Java code to JavaScript:\n\n{java_code}"}
        ]
        
        if context:
            messages.insert(1, {"role": "system", "content": f"Context:\n{context}"})
        
        response = self.client.chat.completions.create(
            model=self.model,
            messages=messages,
            temperature=0.3,
            max_tokens=4000
        )
        
        return response.choices[0].message.content
```

```python
# ai_engine/services/model_router.py
import logging
from typing import Optional
from .modal_client import ModalClient
from .deepseek_client import DeepSeekClient
from .ollama_client import OllamaClient

logger = logging.getLogger(__name__)

class ModelRouter:
    def __init__(self):
        self.primary = ModalClient()  # CodeT5+ on Modal
        self.fallback = DeepSeekClient()  # DeepSeek API
        self.backup = OllamaClient()  # Local Ollama
        self.cost_tracker = CostTracker()
    
    def translate(self, java_code: str, context: list = None) -> str:
        # Try primary (Modal)
        try:
            if self.primary.health_check():
                result = self.primary.translate(java_code, context)
                self.cost_tracker.record("modal", cost=0.05)  # ~$0.05 per conversion
                return result
        except Exception as e:
            logger.warning(f"Primary model failed: {e}, trying fallback")
        
        # Try fallback (DeepSeek)
        try:
            result = self.fallback.translate(java_code, context)
            self.cost_tracker.record("deepseek", cost=0.10)
            return result
        except Exception as e:
            logger.warning(f"Fallback model failed: {e}, trying backup")
        
        # Try backup (Ollama)
        try:
            result = self.backup.translate(java_code)
            self.cost_tracker.record("ollama", cost=0.00)
            return result
        except Exception as e:
            logger.error(f"All models failed: {e}")
            raise RuntimeError("All translation models unavailable")
```

```python
# ai_engine/services/cost_tracker.py
from datetime import datetime, timedelta
from typing import Dict, List

class CostTracker:
    def __init__(self):
        self.costs: List[Dict] = []
        self.model_costs = {
            "modal": 0.05,  # Per conversion
            "deepseek": 0.10,
            "ollama": 0.00,
            "gpt4": 0.50,
        }
    
    def record(self, model: str, cost: float = None):
        if cost is None:
            cost = self.model_costs.get(model, 0.0)
        
        self.costs.append({
            "model": model,
            "cost": cost,
            "timestamp": datetime.utcnow(),
        })
    
    def get_daily_cost(self, date: datetime = None) -> float:
        if date is None:
            date = datetime.utcnow()
        
        return sum(
            c["cost"] for c in self.costs
            if c["timestamp"].date() == date.date()
        )
    
    def get_weekly_cost(self) -> float:
        week_ago = datetime.utcnow() - timedelta(days=7)
        return sum(c["cost"] for c in self.costs if c["timestamp"] >= week_ago)
    
    def get_average_cost_per_conversion(self) -> float:
        if not self.costs:
            return 0.0
        return sum(c["cost"] for c in self.costs) / len(self.costs)
```
