### Task 2.5.6.1: Conversion Metrics Tracking

**Type**: Implementation
**Effort**: High (8 hours)
**Dependencies**: None
**Verification**: Metrics accurately tracked

**Description**: Implement system to track conversion metrics including success rates, failure types, and processing times.

**Acceptance Criteria**:
- [ ] Track conversion success rate (overall and per mode)
- [ ] Track failure types and frequencies
- [ ] Track processing time per conversion
- [ ] Track mod complexity impact on success
- [ ] Track retry success rates

**Implementation**:
```python
class AutomationMetrics:
    """Track automation-related metrics."""
    
    def __init__(self):
        self._metrics = {
            "total_conversions": 0,
            "successful_conversions": 0,
            "failed_conversions": 0,
            "auto_recovered": 0,
            "manual_intervention": 0,
            "by_mode": {},  # Simple/Standard/Complex/Expert
            "by_error_type": {},
            "processing_times": [],
            "retry_success_rate": 0.0,
        }
    
    async def record_conversion(
        self,
        success: bool,
        mode: str,
        processing_time: float,
        error_type: Optional[str] = None,
        auto_recovered: bool = False,
    ) -> None:
        """Record conversion result."""
        self._metrics["total_conversions"] += 1
        if success:
            self._metrics["successful_conversions"] += 1
        else:
            self._metrics["failed_conversions"] += 1
        
        if auto_recovered:
            self._metrics["auto_recovered"] += 1
        elif not success:
            self._metrics["manual_intervention"] += 1
        
        # Track by mode
        if mode not in self._metrics["by_mode"]:
            self._metrics["by_mode"][mode] = {"total": 0, "success": 0}
        self._metrics["by_mode"][mode]["total"] += 1
        if success:
            self._metrics["by_mode"][mode]["success"] += 1
        
        # Track by error type
        if error_type:
            if error_type not in self._metrics["by_error_type"]:
                self._metrics["by_error_type"][error_type] = 0
            self._metrics["by_error_type"][error_type] += 1
        
        # Track processing time
        self._metrics["processing_times"].append(processing_time)
    
    def get_success_rate(self) -> float:
        """Calculate overall success rate."""
        if self._metrics["total_conversions"] == 0:
            return 0.0
        return self._metrics["successful_conversions"] / self._metrics["total_conversions"]
    
    def get_mode_success_rates(self) -> Dict[str, float]:
        """Calculate success rate per mode."""
        return {
            mode: data["success"] / data["total"] if data["total"] > 0 else 0.0
            for mode, data in self._metrics["by_mode"].items()
        }
    
    def get_auto_recovery_rate(self) -> float:
        """Calculate auto-recovery rate."""
        total_failures = self._metrics["failed_conversions"]
        if total_failures == 0:
            return 0.0
        return self._metrics["auto_recovered"] / total_failures
    
    def get_avg_processing_time(self) -> float:
        """Calculate average processing time."""
        times = self._metrics["processing_times"]
        if not times:
            return 0.0
        return sum(times) / len(times)
```

class BottleneckDetector:
    """Detect performance bottlenecks in conversion pipeline."""
    
    STAGE_THRESHOLDS = {
        "parsing": 5.0,      # seconds
        "analysis": 10.0,     # seconds
        "translation": 30.0,  # seconds
        "packaging": 5.0,      # seconds
    }
    
    def __init__(self):
        self._stage_times = defaultdict(list)
        self._bottlenecks = []
    
    async def record_stage_time(
        self,
        conversion_id: str,
        stage: str,
        duration: float,
    ) -> None:
        """Record time for a conversion stage."""
        self._stage_times[stage].append(duration)
    
    def get_bottlenecks(self) -> List[Dict[str, Any]]:
        """Identify current bottlenecks."""
        bottlenecks = []
        for stage, times in self._stage_times.items():
            if not times:
                continue
            avg_time = sum(times) / len(times)
            threshold = self.STAGE_THRESHOLDS.get(stage, 10.0)
            if avg_time > threshold:
                bottlenecks.append({
                    "stage": stage,
                    "avg_time": avg_time,
                    "threshold": threshold,
                    "severity": "high" if avg_time > threshold * 2 else "medium",
                    "recommendation": self._get_recommendation(stage, avg_time),
                })
        return bottlenecks
    
    def get_stage_statistics(self) -> Dict[str, Dict[str, float]]:
        """Get statistics per stage."""
        stats = {}
        for stage, times in self._stage_times.items():
            if not times:
                continue
            sorted_times = sorted(times)
            stats[stage] = {
                "avg": sum(times) / len(times),
                "min": min(times),
                "max": max(times),
                "p50": sorted_times[len(times) // 2],
                "p95": sorted_times[int(len(times) * 0.95)],
                "p99": sorted_times[int(len(times) * 0.99)],
            }
        return stats
```

class TrendAnalyzer:
    """Analyze trends in automation metrics."""
    
    def __init__(self, retention_days: int = 30):
        self._historical_data = []
        self._retention_days = retention_days
    
    async def record_snapshot(
        self,
        metrics: Dict[str, Any],
    ) -> None:
        """Record a metrics snapshot with timestamp."""
        self._historical_data.append({
            "timestamp": datetime.now(),
            "metrics": metrics.copy(),
        })
        self._cleanup_old_data()
    
    def get_success_rate_trend(
        self,
        period: str = "24h",
    ) -> List[Dict[str, Any]]:
        """Get success rate trend over period."""
        cutoff = self._get_cutoff(period)
        snapshots = [
            s for s in self._historical_data
            if s["timestamp"] >= cutoff
        ]
        return [
            {
                "timestamp": s["timestamp"].isoformat(),
                "success_rate": s["metrics"].get("success_rate", 0),
            }
            for s in snapshots
        ]
    
    def detect_anomalies(
        self,
        metric: str,
        threshold: float = 2.0,
    ) -> List[Dict[str, Any]]:
        """Detect anomalies using standard deviation."""
        values = [s["metrics"].get(metric, 0) for s in self._historical_data]
        if len(values) < 10:
            return []
        
        mean = sum(values) / len(values)
        std = (sum((x - mean) ** 2 for x in values) / len(values)) ** 0.5
        
        anomalies = []
        for s in self._historical_data:
            value = s["metrics"].get(metric, 0)
            z_score = abs((value - mean) / std) if std > 0 else 0
            if z_score > threshold:
                anomalies.append({
                    "timestamp": s["timestamp"].isoformat(),
                    "value": value,
                    "z_score": z_score,
                    "deviation": value - mean,
                })
        return anomalies
    
    def get_improvement_recommendations(self) -> List[str]:
        """Generate improvement recommendations based on trends."""
        recommendations = []
        
        # Check success rate trend
        recent_trend = self._calculate_trend("success_rate")
        if recent_trend < -0.05:
            recommendations.append(
                "Success rate is declining. Review recent failure patterns."
            )
        
        # Check processing time trend
        recent_time = self._calculate_trend("avg_processing_time")
        if recent_time > 0.2:
            recommendations.append(
                "Processing time is increasing. Consider scaling resources."
            )
        
        return recommendations
```

class AutomationDashboard:
    """Real-time automation analytics dashboard."""
    
    def __init__(
        self,
        metrics: AutomationMetrics,
        bottleneck_detector: BottleneckDetector,
        trend_analyzer: TrendAnalyzer,
    ):
        self._metrics = metrics
        self._bottleneck_detector = bottleneck_detector
        self._trend_analyzer = trend_analyzer
    
    def get_dashboard_data(self) -> Dict[str, Any]:
        """Get all dashboard data in a single query."""
        return {
            "overview": {
                "total_conversions": self._metrics._metrics["total_conversions"],
                "success_rate": self._metrics.get_success_rate(),
                "auto_recovery_rate": self._metrics.get_auto_recovery_rate(),
                "avg_processing_time": self._metrics.get_avg_processing_time(),
            },
            "by_mode": self._metrics.get_mode_success_rates(),
            "error_types": self._metrics._metrics["by_error_type"],
            "bottlenecks": self._bottleneck_detector.get_bottlenecks(),
            "stage_stats": self._bottleneck_detector.get_stage_statistics(),
            "alerts": self._get_active_alerts(),
        }
    
    def get_realtime_update(self) -> Dict[str, Any]:
        """Get lightweight update for WebSocket."""
        return {
            "success_rate": self._metrics.get_success_rate(),
            "active_conversions": self._get_active_count(),
            "recent_bottlenecks": self._bottleneck_detector.get_bottlenecks()[:3],
            "timestamp": datetime.now().isoformat(),
        }
