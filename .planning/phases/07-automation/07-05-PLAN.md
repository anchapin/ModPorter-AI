```python
class ErrorPatternDetector:
    """Detect and classify conversion errors."""

    ERROR_PATTERNS = {
        ErrorType.SYNTAX: [
            r"unexpected token",
            r"expected .+ but found",
            r"syntax error",
            r"invalid expression",
            r"unterminated string",
        ],
        ErrorType.MISSING_PATTERN: [
            r"cannot find symbol",
            r"undefined reference",
            r"missing resource",
            r"unknown block",
            r"cannot resolve method",
        ],
        ErrorType.TYPE_MISMATCH: [
            r"incompatible types",
            r"type mismatch",
            r"cannot convert",
            r"expected .+ but .+ was",
        ],
        ErrorType.API_INCOMPATIBILITY: [
            r"method not found",
            r"deprecated API",
            r"unsupported operation",
            r"API version mismatch",
        ],
        ErrorType.RESOURCE_ERROR: [
            r"file not found",
            r"resource not found",
            r"IO error",
            r"corrupted file",
        ],
    }

    def __init__(self):
        self.error_history: List[ConversionError] = []
        self.pattern_matcher = PatternMatcher(self.ERROR_PATTERNS)

    async def detect_error(
        self,
        error: Exception,
        context: ErrorContext,
    ) -> ClassifiedError:
        """Detect and classify error from exception."""

        # Extract error message
        error_message = self._extract_message(error)

        # Match against patterns
        matched_pattern = self.pattern_matcher.match(error_message)

        # Extract context
        context_info = await self._extract_context(context)

        # Calculate confidence
        confidence = self._calculate_confidence(
            pattern=matched_pattern,
            context=context_info,
        )

        # Create classified error
        return ClassifiedError(
            error_type=matched_pattern.error_type,
            message=error_message,
            confidence=confidence,
            context=context_info,
            recovery_priority=self._get_recovery_priority(matched_pattern.error_type),
        )

    def _extract_message(self, error: Exception) -> str:
        """Extract clean error message."""
        message = str(error)

        # Clean up stack traces
        if "Exception" in message:
            parts = message.split(":")
            if len(parts) > 1:
                message = parts[-1].strip()

        return message

    async def _extract_context(
        self,
        context: ErrorContext,
    ) -> ErrorContextInfo:
        """Extract additional context for recovery."""

        return ErrorContextInfo(
            file_path=context.current_file,
            line_number=context.error_line,
            column=context.error_column,
            method_name=context.current_method,
            class_name=context.current_class,
            imports=context.available_imports,
            dependencies=context.loaded_dependencies,
            stack_trace=context.stack_trace,
        )

    def _calculate_confidence(
        self,
        pattern: MatchedPattern,
        context: ErrorContextInfo,
    ) -> float:
        """Calculate confidence score for error classification."""

        base_confidence = pattern.confidence

        # Boost confidence with more context
        if context.file_path:
            base_confidence += 0.1
        if context.line_number:
            base_confidence += 0.05
        if context.method_name:
            base_confidence += 0.05

        return min(base_confidence, 1.0)

    def _get_recovery_priority(
        self,
        error_type: ErrorType,
    ) -> RecoveryPriority:
        """Determine recovery priority by error type."""

        priorities = {
            ErrorType.SYNTAX: RecoveryPriority.HIGH,  # Often fixable
            ErrorType.MISSING_PATTERN: RecoveryPriority.HIGH,  # Generic fallback
            ErrorType.TYPE_MISMATCH: RecoveryPriority.MEDIUM,  # Inference possible
            ErrorType.API_INCOMPATIBILITY: RecoveryPriority.MEDIUM,  # Workarounds exist
            ErrorType.RESOURCE_ERROR: RecoveryPriority.LOW,  # External issue
        }

        return priorities.get(error_type, RecoveryPriority.LOW)
```

```python
class AutoRecoveryEngine:
    """Execute automatic recovery strategies."""

    def __init__(self):
        self.strategies: Dict[ErrorType, RecoveryStrategy] = {
            ErrorType.SYNTAX: SyntaxErrorStrategy(),
            ErrorType.MISSING_PATTERN: MissingPatternStrategy(),
            ErrorType.TYPE_MISMATCH: TypeMismatchStrategy(),
            ErrorType.API_INCOMPATIBILITY: APICompatibilityStrategy(),
            ErrorType.RESOURCE_ERROR: ResourceErrorStrategy(),
        }

        self.pattern_library = PatternLibrary()
        self.type_inferrer = TypeInferrer()
        self.api_resolver = APIResolver()

    async def attempt_recovery(
        self,
        classified_error: ClassifiedError,
        conversion: ConversionJob,
    ) -> RecoveryResult:
        """Attempt automatic recovery for classified error."""

        strategy = self.strategies.get(classified_error.error_type)

        if not strategy:
            return RecoveryResult(
                success=False,
                error="No recovery strategy available",
                manual_intervention_required=True,
            )

        # Execute recovery
        try:
            result = await strategy.recover(
                error=classified_error,
                conversion=conversion,
            )

            # Validate recovery
            if result.success:
                validation = await self._validate_recovery(
                    conversion=conversion,
                )

                if validation.valid:
                    return RecoveryResult(
                        success=True,
                        recovery_action=result.action,
                        confidence=result.confidence,
                        recovery_time=result.elapsed_time,
                    )
                else:
                    # Recovery produced invalid output
                    return RecoveryResult(
                        success=False,
                        error=f"Recovery produced invalid output: {validation.errors}",
                        manual_intervention_required=False,
                        retry_recommended=True,
                    )

            return RecoveryResult(
                success=False,
                error=result.error,
                manual_intervention_required=result.requires_manual,
            )

        except Exception as e:
            return RecoveryResult(
                success=False,
                error=str(e),
                manual_intervention_required=False,
                retry_recommended=True,
            )


class SyntaxErrorStrategy:
    """Recover from syntax errors."""

    async def recover(
        self,
        error: ClassifiedError,
        conversion: ConversionJob,
    ) -> StrategyResult:
        """Attempt to fix syntax error."""

        start_time = time.time()

        # Get the problematic code
        code = await self._get_error_code(error.context)

        # Try automated fixes
        fixes = [
            self._fix_missing_semicolon,
            self._fix_unterminated_string,
            self._fix_invalid_token,
            self._fix_mismatched_brackets,
        ]

        for fix in fixes:
            try:
                fixed_code = await fix(code, error.context)

                # Validate fix
                if await self._validate_syntax(fixed_code):
                    return StrategyResult(
                        success=True,
                        action=f"Applied {fix.__name__}",
                        confidence=0.95,
                        elapsed_time=time.time() - start_time,
                        modified=True,
                    )
            except FixNotApplicable:
                continue

        return StrategyResult(
            success=False,
            error="Could not auto-fix syntax error",
            requires_manual=True,
            elapsed_time=time.time() - start_time,
        )

    async def _fix_missing_semicolon(
        self,
        code: str,
        context: ErrorContextInfo,
    ) -> str:
        """Add missing semicolons."""

        lines = code.split("\n")
        error_line = context.line_number - 1  # 0-indexed

        if 0 <= error_line < len(lines):
            line = lines[error_line]

            # Check if line ends with incomplete statement
            if self._is_incomplete_statement(line):
                lines[error_line] = line + ";"

        return "\n".join(lines)

    async def _validate_syntax(self, code: str) -> bool:
        """Validate JavaScript syntax."""

        try:
            ast.parse(code)
            return True
        except SyntaxError:
            return False

class TypeMismatchStrategy:
    """Recover from type mismatch errors."""

    async def recover(
        self,
        error: ClassifiedError,
        conversion: ConversionJob,
    ) -> StrategyResult:
        """Attempt to infer correct type."""

        start_time = time.time()

        # Extract expected and actual types
        expected_type = self._extract_expected_type(error.message)
        actual_type = self._extract_actual_type(error.message)

        # Infer correct type from context
        inferred = await self.type_inferrer.infer(
            variable=error.context.current_method,
            expected=expected_type,
            actual=actual_type,
            code=conversion.current_code,
        )

        if inferred:
            return StrategyResult(
                success=True,
                action=f"Inferred type: {inferred}",
                confidence=0.90,
                elapsed_time=time.time() - start_time,
                modified=True,
            )

        # Fallback to safe type
        return StrategyResult(
            success=True,
            action=f"Applied safe type: any",
            confidence=0.70,
            elapsed_time=time.time() - start_time,
            modified=True,
            warning="Type safety relaxed",
        )

class ResourceErrorStrategy:
    """Recover from resource errors."""

    MAX_RETRIES = 3
    BASE_DELAY = 2  # seconds

    async def recover(
        self,
        error: ClassifiedError,
        conversion: ConversionJob,
    ) -> StrategyResult:
        """Retry with exponential backoff."""

        start_time = time.time()

        for attempt in range(self.MAX_RETRIES):
            try:
                # Attempt to reload resource
                await self._reload_resource(error.context)

                return StrategyResult(
                    success=True,
                    action=f"Resource loaded on attempt {attempt + 1}",
                    confidence=0.99,
                    elapsed_time=time.time() - start_time,
                    modified=False,
                )

            except ResourceNotFoundError:
                # Resource genuinely missing
                if attempt == self.MAX_RETRIES - 1:
                    return StrategyResult(
                        success=False,
                        error="Resource not found after retries",
                        requires_manual=True,
                        elapsed_time=time.time() - start_time,
                    )

                # Wait with backoff
                delay = self.BASE_DELAY * (2 ** attempt)
                await asyncio.sleep(delay)

            except ResourceCorruptedError:
                # Try to regenerate
                await self._regenerate_resource(error.context)

        return StrategyResult(
            success=False,
            error="Resource recovery failed",
            requires_manual=True,
            elapsed_time=time.time() - start_time,
        )
```

**Acceptance Criteria**:

**Fallback Chain**:
```
Auto-Recovery Attempt
       │
       ├─► Success ──► Continue conversion
       │
       ├─► Partial Success ──► Mark for review, continue
       │
       └─► Failure ──► Fallback chain:
                       │
                       ├─► Try alternative strategy
                       │
                       ├─► Apply safe default
                       │
                       ├─► Skip problematic element
                       │
                       └─► Queue for manual review
```

**Implementation**:
```python
class FallbackManager:
    """Manage fallback mechanisms for failed recovery."""

    def __init__(self):
        self.fallback_chain = [
            FallbackStep.ALTERNATIVE_STRATEGY,
            FallbackStep.SAFE_DEFAULT,
            FallbackStep.SKIP_ELEMENT,
            FallbackStep.MANUAL_REVIEW,
        ]

    async def execute_fallback(
        self,
        error: ClassifiedError,
        conversion: ConversionJob,
    ) -> FallbackResult:
        """Execute fallback chain."""

        for step in self.fallback_chain:
            result = await self._execute_step(step, error, conversion)

            if result.success:
                return FallbackResult(
                    step=step,
                    success=True,
                    action=result.action,
                    requires_review=step == FallbackStep.MANUAL_REVIEW,
                )

        # All fallbacks failed
        return FallbackResult(
            step=FallbackStep.MANUAL_REVIEW,
            success=False,
            action="Queued for manual review",
            requires_review=True,
        )

    async def _execute_step(
        self,
        step: FallbackStep,
        error: ClassifiedError,
        conversion: ConversionJob,
    ) -> StepResult:
        """Execute a single fallback step."""

        if step == FallbackStep.ALTERNATIVE_STRATEGY:
            return await self._try_alternative_strategy(error, conversion)

        elif step == FallbackStep.SAFE_DEFAULT:
            return await self._apply_safe_default(error, conversion)

        elif step == FallbackStep.SKIP_ELEMENT:
            return await self._skip_problematic_element(error, conversion)

        elif step == FallbackStep.MANUAL_REVIEW:
            return await self._queue_for_manual_review(error, conversion)

    async def _apply_safe_default(
        self,
        error: ClassifiedError,
        conversion: ConversionJob,
    ) -> StepResult:
        """Apply safe default values."""

        defaults = {
            ErrorType.SYNTAX_ERROR: "// Syntax error placeholder",
            ErrorType.MISSING_PATTERN: "null",
            ErrorType.TYPE_MISMATCH: "any",
            ErrorType.API_INCOMPATIBILITY: "undefined",
            ErrorType.RESOURCE_ERROR: '""',
        }

        default_value = defaults.get(error.error_type, "null")

        return StepResult(
            success=True,
            action=f"Applied safe default: {default_value}",
        )

    async def _skip_problematic_element(
        self,
        error: ClassifiedError,
        conversion: ConversionJob,
    ) -> StepResult:
        """Skip problematic element to preserve partial conversion."""

        # Mark element as skipped
        conversion.skipped_elements.append(SkippedElement(
            reason=error.error_type,
            location=error.context.file_path,
            line=error.context.line_number,
            original_code=error.context.current_method or "",
        ))

        return StepResult(
            success=True,
            action="Skipped problematic element",
            modified=True,
        )

    async def _queue_for_manual_review(
        self,
        error: ClassifiedError,
        conversion: ConversionJob,
    ) -> StepResult:
        """Queue conversion for manual review."""

        review_request = ReviewRequest(
            conversion_id=conversion.id,
            error=error,
            priority=error.recovery_priority,
            created_at=datetime.utcnow(),
        )

        await self.review_queue.enqueue(review_request)

        return StepResult(
            success=True,
            action="Queued for manual review",
        )


class CircuitBreaker:
    """Prevent error cascade with circuit breaker."""

    def __init__(self, failure_threshold: int = 5, timeout: int = 60):
        self.failure_threshold = failure_threshold
        self.timeout = timeout
        self.failures = 0
        self.last_failure_time = None
        self.state = CircuitState.CLOSED

    async def call(self, func: Callable, *args, **kwargs):
        """Execute function with circuit breaker protection."""

        if self.state == CircuitState.OPEN:
            if self._should_attempt_reset():
                self.state = CircuitState.HALF_OPEN
            else:
                raise CircuitBreakerOpenError("Circuit breaker is open")

        try:
            result = await func(*args, **kwargs)

            if self.state == CircuitState.HALF_OPEN:
                self._reset()

            return result

        except Exception as e:
            self._record_failure()

            if self.failures >= self.failure_threshold:
                self.state = CircuitState.OPEN
                self.last_failure_time = datetime.utcnow()

            raise
```

```python
@dataclass
class RecoveryMetrics:
    """Metrics for error auto-recovery."""

    total_errors: int
    auto_recovered: int
    manual_intervention: int
    failed: int

    @property
    def auto_recovery_rate(self) -> float:
        """Calculate auto-recovery rate."""
        if self.total_errors == 0:
            return 1.0
        return self.auto_recovered / self.total_errors

    @property
    def manual_intervention_rate(self) -> float:
        """Calculate manual intervention rate."""
        if self.total_errors == 0:
            return 0.0
        return self.manual_intervention / self.total_errors
```

```python
class RecoveryTracker:
    """Track recovery success and metrics."""

    def __init__(self):
        self.records: List[RecoveryRecord] = []
        self.metrics = RecoveryMetricsStore()

    async def record_attempt(
        self,
        error: ClassifiedError,
        strategy: str,
        result: RecoveryResult,
        recovery_time_ms: int,
    ):
        """Record recovery attempt."""

        record = RecoveryRecord(
            error_id=generate_error_id(),
            error_type=error.error_type,
            recovery_strategy=strategy,
            success=result.success,
            recovery_time_ms=recovery_time_ms,
            fallback_used=result.fallback_used,
            timestamp=datetime.utcnow(),
        )

        self.records.append(record)
        await self._update_metrics(record)

    async def _update_metrics(self, record: RecoveryRecord):
        """Update aggregate metrics."""

        await self.metrics.increment(
            error_type=record.error_type,
            success=record.success,
            recovery_time=record.recovery_time_ms,
        )

    async def get_metrics(
        self,
        time_range: Optional[timedelta] = None,
    ) -> RecoveryMetrics:
        """Get current recovery metrics."""

        records = self._filter_by_time(time_range)

        return RecoveryMetrics(
            total_errors=len(records),
            auto_recovered=sum(1 for r in records if r.success),
            manual_intervention=sum(1 for r in records if r.fallback_used),
            failed=sum(1 for r in records if not r.success and not r.fallback_used),
        )

    async def get_error_patterns(self) -> Dict[ErrorType, int]:
        """Get error frequency by type."""

        patterns = defaultdict(int)

        for record in self.records:
            patterns[record.error_type] += 1

        return dict(patterns)

    async def get_recovery_times(self) -> Dict[str, float]:
        """Get average recovery time by strategy."""

        times = defaultdict(list)

        for record in self.records:
            times[record.recovery_strategy].append(record.recovery_time_ms)

        return {
            strategy: sum(times) / len(times)
            for strategy, times in times.items()
            if times
        }
```

```python
async def verify_manual_intervention():

    test_errors = load_test_errors(100)

    manual_count = 0

    for error in test_errors:
        result = await recovery_engine.attempt_recovery(error)
```
