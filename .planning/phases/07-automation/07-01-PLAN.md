```python
def extract_features(mod_path: str) -> ModFeatures:
    """Extract classification features from mod."""
    features = ModFeatures()
    
    # Count classes
    features.class_count = count_java_classes(mod_path)
    
    # Analyze dependencies
    features.dependencies = analyze_dependencies(mod_path)
    
    # Inventory assets
    features.assets = count_assets(mod_path)
    
    # Detect complex features
    features.has_multiblock = detect_multiblock(mod_path)

    features.has_dimension = detect_dimension(mod_path)
```

```python
CLASSIFICATION_RULES = {
    "Simple": {
        "class_count": (1, 5),
        "dependencies": (0, 2),
        "complex_features": [],
        "automation_target": 0.99,
    },
    "Standard": {
        "class_count": (5, 20),
        "dependencies": (2, 5),
        "complex_features": ["entity", "recipe"],
        "automation_target": 0.95,
    },
    "Complex": {
        "class_count": (20, 50),
        "dependencies": (5, 10),
        "complex_features": ["multiblock", "machine", "custom_ai"],
        "automation_target": 0.85,
    },
    "Expert": {
        "class_count": (50, None),  # No upper limit
        "dependencies": (10, None),
        "complex_features": ["dimension", "worldgen", "biome"],
        "automation_target": 0.70,
    },
}

def classify_mod(features: ModFeatures) -> ClassificationResult:
    """Classify mod into mode based on features."""
    
    # Check for expert features first
    if any(f in features.complex_features 
           for f in CLASSIFICATION_RULES["Expert"]["complex_features"]):
        return ClassificationResult(
            mode="Expert",
            confidence=0.9,
            reason="Expert-level features detected",
        )
    
    # Check complex features
    if any(f in features.complex_features 
           for f in CLASSIFICATION_RULES["Complex"]["complex_features"]):
        return ClassificationResult(
            mode="Complex",
            confidence=0.85,
            reason="Complex features detected",
        )
    
    # Check class count and dependencies
    if (CLASSIFICATION_RULES["Standard"]["class_count"][0] <= 
        features.class_count <= 
        CLASSIFICATION_RULES["Standard"]["class_count"][1]):
        return ClassificationResult(
            mode="Standard",
            confidence=0.9,
            reason="Standard complexity",
        )
    
    # Default to Simple
    return ClassificationResult(
        mode="Simple",
        confidence=0.95,
        reason="Basic mod structure",
    )
```

**Confidence Calculation**:
```python
def calculate_confidence(
    features: ModFeatures,
    mode: str,
) -> float:
    """Calculate classification confidence score."""
    
    confidence = 1.0
    
    # Reduce confidence for borderline cases
    class_range = CLASSIFICATION_RULES[mode]["class_count"]
    if class_range[0] and class_range[1]:
        midpoint = (class_range[0] + class_range[1]) / 2
        distance_from_midpoint = abs(features.class_count - midpoint)
        range_size = class_range[1] - class_range[0]
        confidence -= (distance_from_midpoint / range_size) * 0.2
    
    # Reduce confidence for missing information
    if features.class_count == 0:
        confidence -= 0.3
    if len(features.dependencies) == 0:
        confidence -= 0.1
    
    # Reduce confidence for unknown features
    unknown_features = len(features.unknown_features)
    confidence -= unknown_features * 0.05
    
    return max(0.0, min(1.0, confidence))
```
