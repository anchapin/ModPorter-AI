```python
class BatchUploadHandler:
    """Handle batch mod file uploads."""
    
    MAX_BATCH_SIZE = 100
    MAX_FILE_SIZE = 500 * 1024 * 1024  # 500MB
    SUPPORTED_FORMATS = ['.jar', '.zip', '.tar.gz']
    
    async def upload_batch(
        self,
        files: List[UploadFile],
        user_id: str,
    ) -> BatchUploadResult:
        """Process batch upload request."""
        
        # Validate batch size
        if len(files) > self.MAX_BATCH_SIZE:
            raise BatchSizeError(
                f"Maximum {self.MAX_BATCH_SIZE} files allowed"
            )
        
        # Validate each file
        validated_files = []
        errors = []
        
        for file in files:
            try:
                validated = await self._validate_file(file)
                validated_files.append(validated)
            except ValidationError as e:
                errors.append(ItemError(
                    item_id=file.filename,
                    error=str(e),
                    recoverable=False,
                ))
        
        # Check for duplicates
        validated_files = self._check_duplicates(validated_files, user_id)
        
        # Create batch record
        batch = Batch(
            id=generate_batch_id(),
            user_id=user_id,
            items=[self._create_batch_item(f) for f in validated_files],
            status=BatchStatus.PENDING,
            created_at=datetime.utcnow(),
        )
        
        # Store and return
        await self.batch_storage.save(batch)
        
        return BatchUploadResult(
            batch_id=batch.id,
            total_items=len(batch.items),
            valid_items=len(validated_files),
            errors=errors,
        )
```

**Implementation**:
```python
class IntelligentQueueManager:
    """Manage conversion queue with intelligence."""
    
    def __init__(self):
        self.queue = PriorityQueue()
        self.resource_allocator = ResourceAllocator(
            max_cpu_slots=4,
            max_memory_gb=16,
            max_concurrent=4,
        )
        self.worker_pool = WorkerPool(size=4)
        self._running = False
    
    async def enqueue(
        self,
        item: BatchItem,
        priority: Priority = Priority.NORMAL,
    ):
        """Add item to queue with priority."""
        
        # Analyze resource requirements
        requirements = await self._analyze_requirements(item)
        
        # Check dependencies
        if item.dependencies:
            await self._wait_for_dependencies(item)
        
        # Calculate effective priority
        effective_priority = self._calculate_priority(
            base=priority,
            requirements=requirements,
            wait_time=item.queue_wait_time,
        )
        
        # Enqueue with metadata
        queue_entry = QueueEntry(
            item=item,
            priority=effective_priority,
            requirements=requirements,
            enqueued_at=datetime.utcnow(),
        )
        
        await self.queue.put(queue_entry)
        
        # Update item status
        item.status = ItemStatus.QUEUED
        item.priority = effective_priority
    
    async def _calculate_priority(
        self,
        base: Priority,
        requirements: ResourceRequirements,
        wait_time: timedelta,
    ) -> int:
        """Calculate effective priority considering wait time."""
        
        base_priority = base.value
        
        # Boost priority for long-waiting items
        if wait_time > timedelta(minutes=5):
            base_priority += 10
        elif wait_time > timedelta(minutes=2):
            base_priority += 5
        
        # Boost for smaller, simpler jobs
        if requirements.estimated_time < timedelta(minutes=2):
            base_priority += 5
        
        return min(base_priority, 100)  # Cap at 100
    
    async def process_queue(self):
        """Main queue processing loop."""
        
        while self._running:
            # Get next item if resources available
            entry = await self.queue.peek()
            
            if not entry:
                await asyncio.sleep(1)
                continue
            
            # Check resource availability
            if await self.resource_allocator.can_allocate(entry.requirements):
                entry = await self.queue.get()
                
                # Start processing
                asyncio.create_task(self._process_item(entry))
            
            await asyncio.sleep(0.5)
    
    async def _process_item(self, entry: QueueEntry):
        """Process a single queue entry."""
        
        item = entry.item
        item.status = ItemStatus.PROCESSING
        item.started_at = datetime.utcnow()
        
        try:
            # Allocate resources
            async with self.resource_allocator.allocate(entry.requirements):
                # Run conversion
                result = await self.worker_pool.execute(
                    conversion_pipeline,
                    item.file_path,
                )
            
            # Mark complete
            item.status = ItemStatus.COMPLETED
            item.completed_at = datetime.utcnow()
            item.result_path = result.output_path
            item.progress = 1.0
            
        except Exception as e:
            item.status = ItemStatus.FAILED
            item.error = ItemError(
                item_id=item.item_id,
                error=str(e),
                recoverable=self._is_recoverable(e),
            )
```

**Queue Efficiency Metrics**:
```python
def calculate_queue_efficiency(queue: Queue) -> float:
    """Calculate queue efficiency percentage."""
    
    total_items = len(queue.completed) + len(queue.failed)
    if total_items == 0:
        return 0.0
    
    # Efficiency = (actual throughput / theoretical max) * 100
    # Based on: wait time, processing time, resource utilization
    
    avg_wait_time = sum(i.wait_time for i in queue.completed) / len(queue.completed)
    avg_process_time = sum(i.processing_time for i in queue.completed) / len(queue.completed)
    
    # Ideal: wait_time ≈ 0, all time spent processing
    efficiency = (avg_process_time / (avg_wait_time + avg_process_time)) * 100
    
    return min(efficiency, 100.0)
```

```python
class PriorityProcessor:
    """Handle priority-based processing."""
    
    PRIORITY_TIERS = {
        "vip": 100,
        "high": 75,
        "normal": 50,
        "low": 25,
        "idle": 10,
    }
    
    def get_item_priority(
        self,
        item: BatchItem,
        user_tier: UserTier,
        batch_priority: str,
    ) -> int:
        """Calculate final priority for item."""
        
        # Start with batch priority
        priority = self.PRIORITY_TIERS.get(batch_priority, 50)
        
        # Upgrade based on user tier
        if user_tier == UserTier.VIP:
            priority = max(priority, 90)
        elif user_tier == UserTier.PREMIUM:
            priority = max(priority, 70)
        
        # Apply mode-based boost
        if item.mode_classification in ["Simple", "Standard"]:
            # Faster processing, can use higher priority
            priority += 5
        
        # Apply aging boost
        queue_time = (datetime.utcnow() - item.queued_at).total_seconds()
        if queue_time > 600:  # 10 minutes
            priority += 15
        elif queue_time > 300:  # 5 minutes
            priority += 10
        
        return min(priority, 100)
    
    async def process_by_priority(self):
        """Process items respecting priority."""
        
        while True:
            # Get highest priority item
            item = await self.queue.get_highest_priority()
            
            if item:
                await self.process_item(item)
            
            await asyncio.sleep(0.1)
```

**Error Types & Recovery**:
| Error Type | Recovery Strategy | Retry Limit |
|------------|-------------------|-------------|
| **Syntax Error** | Auto-fix + validate | 2 |
| **Missing Dependency** | Fallback resolution | 1 |
| **API Timeout** | Retry with backoff | 3 |
| **Invalid Format** | Skip (non-recoverable) | 0 |
| **Resource Exhausted** | Queue for later | 5 |
| **Unknown Error** | Log + skip | 1 |

**Implementation**:
```python
class ItemErrorHandler:
    """Handle errors for individual batch items."""
    
    MAX_RETRIES = {
        ErrorType.SYNTAX: 2,
        ErrorType.DEPENDENCY: 1,
        ErrorType.TIMEOUT: 3,
        ErrorType.RESOURCE: 5,
        ErrorType.UNKNOWN: 1,
    }
    
    RECOVERY_STRATEGIES = {
        ErrorType.SYNTAX: self._recover_syntax,
        ErrorType.DEPENDENCY: self._recover_dependency,
        ErrorType.TIMEOUT: self._recover_timeout,
        ErrorType.RESOURCE: self._requeue_resource,
    }
    
    async def handle_error(
        self,
        item: BatchItem,
        error: Exception,
    ) -> ErrorHandlingResult:
        """Handle item error with recovery."""
        
        error_type = self._classify_error(error)
        item.error = ItemError(
            item_id=item.item_id,
            error_type=error_type,
            message=str(error),
            timestamp=datetime.utcnow(),
            retry_count=item.retry_count,
        )
        
        # Check if recoverable
        if not self._is_recoverable(error_type):
            item.status = ItemStatus.FAILED
            return ErrorHandlingResult(
                action=ErrorAction.FAILED,
                recoverable=False,
            )
        
        # Check retry limit
        if item.retry_count >= self.MAX_RETRIES.get(error_type, 0):
            item.status = ItemStatus.FAILED
            return ErrorHandlingResult(
                action=ErrorAction.FAILED,
                recoverable=False,
                reason="Max retries exceeded",
            )
        
        # Try recovery
        recovery_strategy = self.RECOVERY_STRATEGIES.get(error_type)
        if recovery_strategy:
            try:
                await recovery_strategy(item, error)
                item.retry_count += 1
                item.status = ItemStatus.QUEUED
                
                return ErrorHandlingResult(
                    action=ErrorAction.RETRYING,
                    recoverable=True,
                    retry_after=item.retry_count * 2,  # Exponential backoff
                )
            except Exception as recovery_error:
                # Recovery failed, log and continue
                pass
        
        # Default: requeue with backoff
        await self._requeue_with_backoff(item)
        
        return ErrorHandlingResult(
            action=ErrorAction.REQUEUED,
            recoverable=True,
        )
    
    async def _recover_syntax(self, item: BatchItem, error: Exception):
        """Attempt syntax error recovery."""
        
        # Try auto-fix
        result = await self.syntax_fixer.try_fix(item.file_path)
        
        if result.fixed:
            # Validate fix
            validation = await self.validator.validate(result.output)
            if validation.valid:
                item.file_path = result.output
                return
        
        raise RecoveryFailedError("Could not auto-fix syntax")
    
    async def _requeue_with_backoff(self, item: BatchItem):
        """Requeue item with exponential backoff."""
        
        backoff_seconds = (2 ** item.retry_count) * 10  # 20s, 40s, 80s...
        
        # Schedule retry
        item.status = ItemStatus.QUEUED
        item.retry_at = datetime.utcnow() + timedelta(seconds=backoff_seconds)
        
        await self.queue.schedule(item, delay=backoff_seconds)
```
